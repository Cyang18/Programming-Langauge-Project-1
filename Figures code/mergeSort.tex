\usepackage{algorithm}
\usepackage{algpseudocode}

\begin{algorithm}
\caption{Merge Sort}
\begin{algorithmic}[1]
\Procedure{MergeSort}{$arr$, $left$, $right$}
    \If{$left < right$}
        \State $mid \gets \lfloor(left + right) / 2\rfloor$
        \State \Call{MergeSort}{$arr$, $left$, $mid$}
        \State \Call{MergeSort}{$arr$, $mid + 1$, $right$}
        \State \Call{Merge}{$arr$, $left$, $mid$, $right$}
    \EndIf
\EndProcedure

\Procedure{Merge}{$arr$, $left$, $mid$, $right$}
    \State $i \gets left$
    \State $j \gets mid + 1$
    \State $k \gets 0$
    \State $temp \gets$ new array of size $right - left + 1$

    \While{$i \leq mid$ \textbf{and} $j \leq right$}
        \If{$arr[i] < arr[j]$}
            \State $temp[k] \gets arr[i]$
            \State $i \gets i + 1$
        \Else
            \State $temp[k] \gets arr[j]$
            \State $j \gets j + 1$
        \EndIf
        \State $k \gets k + 1$
    \EndWhile

    \While{$i \leq mid$}
        \State $temp[k] \gets arr[i]$
        \State $i \gets i + 1$
        \State $k \gets k + 1$
    \EndWhile

    \While{$j \leq right$}
        \State $temp[k] \gets arr[j]$
        \State $j \gets j + 1$
        \State $k \gets k + 1$
    \EndWhile

    \For{$i \gets 0$ \textbf{to} $k - 1$}
        \State $arr[left + i] \gets temp[i]$
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}
