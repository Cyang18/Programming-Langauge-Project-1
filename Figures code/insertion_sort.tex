\usepackage{algorithm}
\usepackage{algpseudocode}

\begin{algorithm}
\caption{Insertion Sort}
\begin{algorithmic}[1]
\Procedure{InsertionSort}{$arr$, $size$} 
    \For{$i \gets 1$ \textbf{to} $size-1$} 
        \State $current \gets arr[i]$
        \State $temp \gets i - 1$
        \While{$temp \geq 0$ \textbf{and} $arr[temp] > current$} 
            \State $arr[temp + 1] \gets arr[temp]$
            \State $temp \gets temp - 1$
        \EndWhile
        \State $arr[temp + 1] \gets current$ 
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}
